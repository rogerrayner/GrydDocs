Setup R√°pido

1. Instalar via NuGet

Install-Package GrydAuthentication.Framework
Install-Package GrydAuthentication.Infrastructure.Auth0

2. Configurar no Program.cs

// Add framework services
builder.Services.AddGrydAuthentication(builder.Configuration);
builder.Services.AddAuth0GrydAuthServices(builder.Configuration);

// Configure middleware
app.UseAuthentication();
app.UseAuthorization();

3. Configurar appsettings.json

{
  "JwtSettings": {
    "SecretKey": "YourSuperSecretKey32CharactersLong",

    "Audience": "GrydAuth",
    "ExpirationMinutes": 60
  },
  "Auth0": {
    "Domain": "your-domain.auth0.com",
    "ClientId": "your-client-id",
    "ClientSecret": "your-client-secret"
  }
}
