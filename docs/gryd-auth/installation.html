Generate RSA Keys

Gere o par de chaves RSA para criptografia dos refresh tokens.

Using OpenSSL

# Generate private key (2048 bits)
openssl genpkey -algorithm RSA -out private_key.pem -pkcs8 -aes256

# Generate public key from private key
openssl rsa -pubout -in private_key.pem -out public_key.pem

# View keys content
cat private_key.pem
cat public_key.pem

Using PowerShell (Windows)

# Generate RSA key pair
$rsa = [System.Security.Cryptography.RSA]::Create(2048)

# Export private key
$privateKey = [System.Convert]::ToBase64String($rsa.ExportRSAPrivateKey())
$privateKeyPem = "-----BEGIN PRIVATE KEY-----`n" + $privateKey + "`n-----END PRIVATE KEY-----"

# Export public key  
$publicKey = [System.Convert]::ToBase64String($rsa.ExportRSAPublicKey())
$publicKeyPem = "-----BEGIN PUBLIC KEY-----`n" + $publicKey + "`n-----END PUBLIC KEY-----"

# Save to files
$privateKeyPem | Out-File -FilePath "private_key.pem" -Encoding utf8
$publicKeyPem | Out-File -FilePath "public_key.pem" -Encoding utf8
