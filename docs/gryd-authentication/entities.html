1. BaseEntity

Classe base abstrata que define propriedades comuns para todas as entidades.

using System.ComponentModel.DataAnnotations;

namespace GrydAuthenticationFramework.Domain.Entities.Common
{
    public abstract class BaseEntity
    {
        [Key]
        public Guid Id { get; protected set; } = Guid.NewGuid();

        public DateTime CreatedAt { get; protected set; } = DateTime.UtcNow;

        public DateTime? UpdatedAt { get; protected set; }

        public string? CreatedBy { get; protected set; }

        public string? UpdatedBy { get; protected set; }

        public bool IsDeleted { get; protected set; } = false;

        public DateTime? DeletedAt { get; protected set; }

        public string? DeletedBy { get; protected set; }

        // Eventos de dom√≠nio
        private readonly List<IDomainEvent> _domainEvents = new();
        public IReadOnlyCollection<IDomainEvent> DomainEvents => _domainEvents.AsReadOnly();

        protected void AddDomainEvent(IDomainEvent domainEvent)
        {
            _domainEvents.Add(domainEvent);
        }

        protected void ClearDomainEvents()
        {
            _domainEvents.Clear();
        }

        public void MarkAsUpdated(string? updatedBy = null)
        {
            UpdatedAt = DateTime.UtcNow;
            UpdatedBy = updatedBy;
        }

        public void MarkAsDeleted(string? deletedBy = null)
        {
            IsDeleted = true;
            DeletedAt = DateTime.UtcNow;
            DeletedBy = deletedBy;
        }

        public void Restore()
        {
            IsDeleted = false;
            DeletedAt = null;
            DeletedBy = null;
        }
    }
}

2. User Entity

Entidade principal que representa um usu√°rio do sistema.

using System.ComponentModel.DataAnnotations;
using GrydAuthenticationFramework.Domain.Events;

namespace GrydAuthenticationFramework.Domain.Entities
{
    public class User : BaseEntity
    {
        private readonly List<UserRole> _userRoles = new();
        private readonly List<RefreshToken> _refreshTokens = new();
        private readonly List<UserSession> _sessions = new();

        protected User() { } // EF Constructor

        public User(string email, string firstName = null, string lastName = null)
        {
            if (string.IsNullOrWhiteSpace(email))
                throw new ArgumentException("Email n√£o pode ser vazio", nameof(email));

            if (!IsValidEmail(email))
                throw new ArgumentException("Email deve ter formato v√°lido", nameof(email));

            Email = email.ToLowerInvariant();
            FirstName = firstName?.Trim();
            LastName = lastName?.Trim();
            IsActive = true;
            EmailConfirmed = false;
            TwoFactorEnabled = false;

            AddDomainEvent(new UserCreatedEvent(Id, Email));
        }

        [Required]
        [StringLength(255)]
        public string Email { get; private set; } = string.Empty;

        [StringLength(50)]
        public string? FirstName { get; private set; }

        [StringLength(50)]
        public string? LastName { get; private set; }

        public string FullName => $"{FirstName} {LastName}".Trim();

        [StringLength(20)]
        public string? PhoneNumber { get; private set; }

        public bool EmailConfirmed { get; private set; }

        public bool PhoneNumberConfirmed { get; private set; }

        public bool IsActive { get; private set; }

        public bool TwoFactorEnabled { get; private set; }

        public DateTime? LastLoginAt { get; private set; }

        public int LoginAttempts { get; private set; }

        public DateTime? LastLoginAttemptAt { get; private set; }

        public DateTime? LockoutEndTime { get; private set; }

        public string? Avatar { get; private set; }

        public string? Auth0UserId { get; private set; }

        public Dictionary<string, object> Metadata { get; private set; } = new();

        // Navigation Properties
        public IReadOnlyCollection<UserRole> UserRoles => _userRoles.AsReadOnly();
        public IReadOnlyCollection<RefreshToken> RefreshTokens => _refreshTokens.AsReadOnly();
        public IReadOnlyCollection<UserSession> Sessions => _sessions.AsReadOnly();

        // Computed Properties
        public bool IsLocked => LockoutEndTime.HasValue && LockoutEndTime > DateTime.UtcNow;
        public List<string> RoleNames => _userRoles.Where(ur => ur.Role.IsActive).Select(ur => ur.Role.Name).ToList();

        // Business Methods
        public void UpdateEmail(string newEmail)
        {
            if (string.IsNullOrWhiteSpace(newEmail))
                throw new ArgumentException("Email n√£o pode ser vazio", nameof(newEmail));

            if (!IsValidEmail(newEmail))
                throw new ArgumentException("Email deve ter formato v√°lido", nameof(newEmail));

            var oldEmail = Email;
            Email = newEmail.ToLowerInvariant();
            EmailConfirmed = false;
            MarkAsUpdated();

            AddDomainEvent(new UserEmailChangedEvent(Id, oldEmail, Email));
        }

        public void UpdateProfile(string? firstName, string? lastName, string? phoneNumber)
        {
            FirstName = firstName?.Trim();
            LastName = lastName?.Trim();
            PhoneNumber = phoneNumber?.Trim();
            MarkAsUpdated();

            AddDomainEvent(new UserProfileUpdatedEvent(Id));
        }

        public void ConfirmEmail()
        {
            if (EmailConfirmed) return;

            EmailConfirmed = true;
            MarkAsUpdated();

            AddDomainEvent(new UserEmailConfirmedEvent(Id, Email));
        }

        public void ConfirmPhoneNumber()
        {
            if (PhoneNumberConfirmed) return;

            PhoneNumberConfirmed = true;
            MarkAsUpdated();
        }

        public void Activate()
        {
            if (IsActive) return;

            IsActive = true;
            MarkAsUpdated();

            AddDomainEvent(new UserActivatedEvent(Id));
        }

        public void Deactivate()
        {
            if (!IsActive) return;

            IsActive = false;
            MarkAsUpdated();

            AddDomainEvent(new UserDeactivatedEvent(Id));
        }

        public void EnableTwoFactor()
        {
            if (TwoFactorEnabled) return;

            TwoFactorEnabled = true;
            MarkAsUpdated();

            AddDomainEvent(new UserTwoFactorEnabledEvent(Id));
        }

        public void DisableTwoFactor()
        {
            if (!TwoFactorEnabled) return;

            TwoFactorEnabled = false;
            MarkAsUpdated();

            AddDomainEvent(new UserTwoFactorDisabledEvent(Id));
        }

        public void RecordLogin()
        {
            LastLoginAt = DateTime.UtcNow;
            LoginAttempts = 0;
            LastLoginAttemptAt = null;
            MarkAsUpdated();

            AddDomainEvent(new UserLoggedInEvent(Id));
        }

        public void RecordFailedLogin()
        {
            LoginAttempts++;
            LastLoginAttemptAt = DateTime.UtcNow;

            // Bloquear ap√≥s 5 tentativas
            if (LoginAttempts >= 5)
            {
                LockoutEndTime = DateTime.UtcNow.AddMinutes(30);
                AddDomainEvent(new UserLockedOutEvent(Id, LockoutEndTime.Value));
            }

            MarkAsUpdated();
        }

        public void Unlock()
        {
            if (!IsLocked) return;

            LockoutEndTime = null;
            LoginAttempts = 0;
            MarkAsUpdated();

            AddDomainEvent(new UserUnlockedEvent(Id));
        }

        public void AddRole(Role role)
        {
            if (role == null)
                throw new ArgumentNullException(nameof(role));

            if (_userRoles.Any(ur => ur.RoleId == role.Id))
                return;

            var userRole = new UserRole(Id, role.Id);
            _userRoles.Add(userRole);
            MarkAsUpdated();

            AddDomainEvent(new UserRoleAddedEvent(Id, role.Id));
        }

        public void RemoveRole(Guid roleId)
        {
            var userRole = _userRoles.FirstOrDefault(ur => ur.RoleId == roleId);
            if (userRole == null) return;

            _userRoles.Remove(userRole);
            MarkAsUpdated();

            AddDomainEvent(new UserRoleRemovedEvent(Id, roleId));
        }

        public void SetAuth0UserId(string auth0UserId)
        {
            Auth0UserId = auth0UserId;
            MarkAsUpdated();
        }

        public void UpdateAvatar(string? avatarUrl)
        {
            Avatar = avatarUrl;
            MarkAsUpdated();
        }

        public void UpdateMetadata(Dictionary<string, object> metadata)
        {
            Metadata = metadata ?? new Dictionary<string, object>();
            MarkAsUpdated();
        }

        private static bool IsValidEmail(string email)
        {
            try
            {
                var addr = new System.Net.Mail.MailAddress(email);
                return addr.Address == email;
            }
            catch
            {
                return false;
            }
        }
    }
}

using System.ComponentModel.DataAnnotations;
using GrydAuthenticationFramework.Domain.Events;

namespace GrydAuthenticationFramework.Domain.Entities
{
    public class Role : BaseEntity
    {

        private readonly List<RolePermission> _rolePermissions = new();

        protected Role() { } // EF Constructor

        public Role(string name, string displayName, string? description = null)

        public List<string> PermissionNames => _rolePermissions
            .Where(rp => rp.Permission.IsActive)
            .Select(rp => rp.Permission.Name)
            .ToList();

        // Business Methods
        public void UpdateInfo(string displayName, string? description)
        {
            if (string.IsNullOrWhiteSpace(displayName))
                throw new ArgumentException("Nome de exibi√ß√£o n√£o pode ser vazio", nameof(displayName));

            DisplayName = displayName.Trim();
            Description = description?.Trim();
            MarkAsUpdated();

            AddDomainEvent(new RoleUpdatedEvent(Id));
        }

        public void Activate()
        {
            if (IsActive) return;

            IsActive = true;
            MarkAsUpdated();

            AddDomainEvent(new RoleActivatedEvent(Id));
        }

        public void Deactivate()
        {
            if (!IsActive) return;

            if (IsSystemRole)
                throw new InvalidOperationException("N√£o √© poss√≠vel desativar role do sistema");

            IsActive = false;
            MarkAsUpdated();

            AddDomainEvent(new RoleDeactivatedEvent(Id));
        }

        public void MarkAsSystemRole()

            IsSystemRole = true;
            MarkAsUpdated();
        }

        public void AddPermission(Permission permission)
        {
            if (permission == null)
                throw new ArgumentNullException(nameof(permission));

            if (_rolePermissions.Any(rp => rp.PermissionId == permission.Id))
                return;

            var rolePermission = new RolePermission(Id, permission.Id);
            _rolePermissions.Add(rolePermission);
            MarkAsUpdated();

            AddDomainEvent(new RolePermissionAddedEvent(Id, permission.Id));
        }

        public void RemovePermission(Guid permissionId)
        {
            var rolePermission = _rolePermissions.FirstOrDefault(rp => rp.PermissionId == permissionId);
            if (rolePermission == null) return;

            _rolePermissions.Remove(rolePermission);
            MarkAsUpdated();

            AddDomainEvent(new RolePermissionRemovedEvent(Id, permissionId));
        }

        public bool HasPermission(string permissionName)
        {
            return _rolePermissions.Any(rp => 
                rp.Permission.Name == permissionName && 
                rp.Permission.IsActive);
        }

        public override string ToString() => $"{DisplayName} ({Name})";
    }
}

using System.ComponentModel.DataAnnotations;

namespace GrydAuthenticationFramework.Domain.Entities
{
    public class RefreshToken : BaseEntity
    {
        protected RefreshToken() { } // EF Constructor

        public RefreshToken(Guid userId, string token, DateTime expiresAt)
        {
            UserId = userId;
            Token = token;
            ExpiresAt = expiresAt;
            IsActive = true;
        }

        public Guid UserId { get; private set; }

        [Required]
        public string Token { get; private set; } = string.Empty;

        public DateTime ExpiresAt { get; private set; }

        public bool IsActive { get; private set; }

        public DateTime? RevokedAt { get; private set; }

        public string? RevokedByIp { get; private set; }

        public string? ReplacedByToken { get; private set; }

        // Navigation Properties
        public User User { get; set; } = null!;

        // Computed Properties
        public bool IsExpired => DateTime.UtcNow >= ExpiresAt;
        public bool IsRevoked => RevokedAt != null;
        public bool IsValid => IsActive && !IsExpired && !IsRevoked;

        // Business Methods
        public void Revoke(string? byIp = null, string? replacedByToken = null)
        {
            RevokedAt = DateTime.UtcNow;
            RevokedByIp = byIp;
            ReplacedByToken = replacedByToken;
            IsActive = false;
            MarkAsUpdated();
        }
    }
}
